y = \sin(x)

y = \csc(x)

y = \cos(x)

y=\left\{0 \le x \le 2\pi : \cos(x)\right\} x = 2\pi

y = \sec(x)

y=\left\{0 \le x \le 2\pi : \sec(x)\right\} x = 2\pi

y=\left\{-\frac{1}{2}\pi\le x\le\frac{1}{2}\pi:\tan\left(x\right)\right\} x=-\frac{1}{2}\pi x=\frac{1}{2}\pi
